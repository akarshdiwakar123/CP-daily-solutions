#include <bits/stdc++.h>
using namespace std;

int main() {
    string s; cin >> s;
    int n = s.size();
    int max_count = 1, count = 1;
    for(int i=1;i<= n-1;i++) {
        if (s[i] == s[i-1]) count++;
        else {
            max_count = max(max_count, count);
            count = 1;
        }
    }
    cout << max(max_count, count) << endl;
}
