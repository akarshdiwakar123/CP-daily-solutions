#include<iostream>
using namespace std;
long calculateways(int K){
long totalways=((long)K*K*(K*K-1))/2;
long  attackingWays = 4 * (K - 1) * (K - 2);
long ans = totalways - attackingWays;
return ans;
}
int main()
{
    int N;
    cin>>N;
    for(int K=1;K<=N;K++ ){
        cout<<calculateways(K)<<" ";
    }
    return 0;
}
